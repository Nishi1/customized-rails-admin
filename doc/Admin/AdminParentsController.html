def send_email(data, user_type)
  if data.present? && user_type.present?
    new_data=Hash.new
    data.each do |k,v|
      new_data[k.to_s]=v.to_s
    end
    if user_type == 1
        AdminMailer.coach_account_creation(new_data).deliver_later        
     else 
        AdminMailer.player_account_creation(new_data).deliver_later      
     end    
   end
end
