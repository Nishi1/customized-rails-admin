def save_university
	if request.xhr?
		response= {'status' => false, 'message'=>I18n.t('model_data.common.error'), 'data'=>'' }
		if params.present?
			@model_data = @@model_name.constantize.new(model_params)
  		if @model_data.save
  			 @universities = University.select("id","name").where("status = ?", true)
  			 html_data = render_to_string(partial: 'get_updated_universities_list', locals: {universities: @universities, selected_university_id: @model_data.id})

		 response = { 'status' => true,'message'=>I18n.t('model_data.create.success', model_name: @@model_name.singularize), 'data' => html_data }
  		else
		response = { 'status' => false,'message'=>I18n.t('model_data.common.error', model_name: @@model_name.singularize), 'data' => @model_data.errors }
		flash[:error] = I18n.t('model_data.common.error')
  		end
		end
		render :json => response	
	else
		redirect_to root_url and return
	end
end

def update
	if params.present? && params[:id].present?
     @model_data = @@model_name.constantize.find(params[:id])
     if @model_data.update_attributes(model_params)
       flash[:notice] = I18n.t 'model_data.update.success', model_name: @@model_name.singularize
       redirect_to send("admin_#{controller_name}_path")
     else	
       @page_title= "Edit "+@@model_name

       render "/admin/#{controller_name}/edit" and return     
     end      
	else
		flash[:warning] = I18n.t('model_data.common.error')
		redirect_to send("admin_#{controller_name}_path")
	end
end
