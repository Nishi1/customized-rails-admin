def index
if current_admin.present?
	@coaches_count = User.where("user_type_id =?", COACH_ID).count
	@players_count = User.where("user_type_id =?", PLAYER_ID).count
	@groups_count = Group.count
	@universities_count = University.count
else
	redirect_to new_admin_session_path
end
